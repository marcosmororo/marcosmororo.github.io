Ver ONUs

Troubleshooting
# ONUs descobertas
show gpon onu unconfigured              # Lista ONUs não provisionadas

# Buscar ONU por serial
show gpon onu | include ZTEGC4B0BCE2    # Filtra ONU pelo serial

# Status geral da PON
show gpon interface 1/1/6               # Status da porta GPON

# Status das ONUs
show gpon onu state                     # Estado das ONUs (online/offline)
show gpon onu detail 1/1/6 1            # Detalhes de uma ONU

# Potência óptica
show gpon onu optical-info 1/1/6 1      # RX/TX da ONU
show gpon interface optics 1/1/6        # Potência da porta PON

# VLAN

# Configuração GPON
show running-config gpon                # Configuração GPON
Reboot ONU
# Reiniciar ONU
gpon onu reset 1/1/1 1                  # Reset ONU na porta/posição
Configuração VLAN
# Criar VLAN
vlan 4094 name GERENCIA                # VLAN de gerência
vlan 2001 name PPPOE                   # VLAN de clientes

# Associar VLAN à porta uplink
interface ethernet 1/1/12
switchport mode trunk
switchport trunk allowed vlan add 2001,4094
Serviço
# Criar serviço (bridge)
service 1 vlan 2001                    # Serviço para clientes
service 2 vlan 4094                    # Serviço de gerência
Interface L3
# Interface de gerência
interface vlan 4094
ip address 10.0.0.1/24                # IP da OLT
no shutdown
Bandwidth Profile
# Criar perfil de banda
gpon profile bandwidth 1G
max 1000000                           # Banda máxima (aprox 1Gb)
Line Profile
# Criar line profile
gpon profile line CLIENTE_PADRAO
tcont 1 bandwidth 1G
gem 1
mapping vlan 2001                     # Mapeamento VLAN
Adicionar ONU
# Adicionar ONU untagged
gpon onu add 1/1/1 1 serial DACM0000115C
name ONU1-A
line-profile CLIENTE_PADRAO

# VLAN untagged
gpon onu vlan 1/1/1 1 ethernet 1 untagged 2001
Adicionar ONU Tagged
# ONU em modo bridge/tagged
gpon onu add 1/1/1 2 serial DACM0000115C
name ONU1-B
line-profile CLIENTE_PADRAO
Service Port
# Associar ONU ao serviço
gpon service-port 1 interface 1/1/1 onu 1 gem 1 vlan 2001
Liberar IP
# Liberação de tráfego IP
ip anti-spoofing disable              # Desativa bloqueio (ou ajustar conforme política)
Isolamento ONU
# Isolar ONU (anti-rogue)
gpon onu isolate 1/1/X Y              # Isola ONU

# Remover isolamento
no gpon onu isolate 1/1/X Y           # Remove isolamento
